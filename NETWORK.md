 * @param {Object} requestData - 请求的业务数据 (如果是空请求，传 {})
 * @param {String} openid - 用户的 openid
 * @returns {Object} 包含 Sign、Ticks 等所有必需字段的 Headers 对象
 */
function generateRequestHeaders(requestData = {}, openid = "") {
    // 1. 生成 Ticks (13位时间戳) 和 Nonce (随机数，模拟源码的 Math.random 逻辑)
    const ticks = String(new Date().getTime());
    const nonce = String(Math.floor(Math.random() * 99999999) + 1);

    // 2. 拷贝业务参数，并去除字符串值的首尾空格
    let c = Object.assign({}, requestData);
    for (let key in c) {
        if (typeof c[key] === 'string') {
            c[key] = c[key].trim();
        }
    }

    // 3. 对业务参数的 Key 按首字母 ASCII 码排序
    let keys = Object.keys(c).sort(function(e, r) {
        return e.charCodeAt(0) - r.charCodeAt(0);
    });

    // 4. 拼接业务参数字符串 g
    let g = "";
    for (let i = 0; i < keys.length; i++) {
        let key = keys[i];
        let val = c[key];
        // 如果是 null、数组或对象，只拼接键名；否则拼接 键名+键值
        if (val == null || Array.isArray(val) || typeof val === 'object') {
            g += key; 
        } else {
            g += key + String(val);
        }
    }

    // 5. 拼接终极长串 (AppId + AppSecret + Ticks + Nonce + g)
    let rawStr = APP_ID + APP_SECRET + ticks + nonce + g;

    // 6. 核心加密逻辑：将所有字符打散，全局按 ASCII 码从小到大排序
    let charArray = rawStr.split("");
    charArray.sort(function(e, r) {
        return e.charCodeAt(0) - r.charCodeAt(0);
    });

    // 7. 组合并去除可能的空白字符
    let sortedStr = charArray.join("").replace(/\s*/g, "");

    // 8. 计算 MD5 并转为大写
    let sign = crypto.createHash('md5').update(sortedStr, 'utf8').digest('hex').toUpperCase();

    // 9. 返回完整的 HTTP Headers 对象
    return {
        "AppId": APP_ID,
        "Sign": sign,
        "xweb_xhr": "1",
        "Ticks": ticks,
        "Nonce": nonce,
        "openid": openid,
        "Content-Type": "application/json;charset=UTF-8"
    };
}
